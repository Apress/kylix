function PopString(var S: string; const Delimiter: string): string;
// Returns the chars to the left of the Delimiter (or whole string);
// sets S to chars to right of Delimiter (or '')
var
  Index: integer;
begin
  Index := FirstMatch(S, Delimiter);
  Result := Copy(S, 1, Index - 1);             // Return leftmost
  Delete(S, 1, Index + Length(Delimiter) - 1); // Remove from S
end; // PopString
